# Locale

```shell
export LC_ALL="en_US.UTF-8"         # Overrides all other LC_* variables and LANG. Sets all categories.
export LANGUAGE="en_US"             # Priority list of languages for messages.
export LANG="enUS.UTF-8"            # Default locale for all categories unless overridden.
export LC_ADDRESS="en_US.UTF-8"     # Formatting of addresses.
export LC_COLLATE="en_US.UTF-8"     # Collation (sort order) of strings.
export LC_CTYPE="en_US.UTF-8"       # Character classification and case conversion.
export LC_IDENTIFICATION="en_US.UTF-8" # Metadata about the locale information.
export LC_MEASUREMENT="en_US.UTF-8" # Measurement units (metric/imperial).
export LC_MESSAGES="en_US.UTF-8"    # Language and format of system messages.
export LC_MONETARY="en_US.UTF-8"    # Formatting of monetary values.
export LC_NAME="en_US.UTF-8"        # Formatting of personal names.
export LC_NUMERIC="en_US.UTF-8"     # Formatting of non-monetary numeric values.
export LC_PAPER="en_US.UTF-8"       # Default paper size.
export LC_TELEPHONE="en_US.UTF-8"   # Formatting of telephone numbers.
export LC_TIME="en_US.UTF-8"        # Formatting of date and time values.
```
